        if (isset($_GET['product_id'])) {

            // Fetch product details based on the product ID
            $query = "SELECT * FROM products WHERE product_id = ?";
            $stmt = $sqliconn->prepare($query);
            $stmt->bind_param("i", $product_id);
            $stmt->execute();
            $result = $stmt->get_result();

            if ($result && $result->num_rows > 0) {
                // Display detailed information about the product
                $product = $result->fetch_assoc();
                echo '<h1>' . $product['product_name'] . '</h1>';
                echo '<p>' . $product['product_info'] . '</p>';
                echo '<p>Price: $' . $product['price'] . '</p>';
                echo '<p>Stock: ' . $product['stock'] . '</p>';
                echo '<img src="' . $product['img'] . '" alt="' . $product['product_name'] . '">';
            } else {
                echo '<p>Product not found.</p>';
            }

            // Close the database connection
            $stmt->close();
            $sqliconn->close();
